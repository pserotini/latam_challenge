55 def get_rate_from_column(self, data, column): 

56 delays = {} 

57 for _, row in data.iterrows(): 

58 if row["delay"] == 1: 

59 if row[column] not in delays: 

60 delays[row[column]] = 1 

61 else: 

62 delays[row[column]] += 1 

63 total = data[column].value_counts().to_dict() 

64 

65 rates = {} 

66 for name, total in total.items(): 

67 if name in delays: 

68 rates[name] = round(total / delays[name], 2) 

69 else: 

70 rates[name] = 0 

71 

72 return pd.DataFrame.from_dict(data = rates, orient = "index", columns = ["Tasa (%)"])
